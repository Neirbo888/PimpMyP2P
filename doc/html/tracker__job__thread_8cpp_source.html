81 void TrackerJobThread::handlePeerRefresh(const PimpMessage &request)
82 {
83  const juce::IPAddress peerIP = request.getSource();
84 
85  if (peerIP.toString() != "0.0.0.0")
86  {
87  _owner->getFileManager().registerPeer(peerIP);
88  PimpMessage acknowledge = PimpMessage::createOk();
89  acknowledge.sendToSocket(_socket, _owner->getLocalIp());
90  if (request.hasLocalFileList())
91  {
92  _owner->getFileManager().cleanPeer(peerIP);
93  juce::Array<PeerFile>& localFiles = _owner->getFileManager().getAvailableFiles();
94  juce::Array<PeerFile> distantFiles = request.getLocalFileList();
95  Logger::writeToLog("Peer has " + juce::String(distantFiles.size()) + " files");
96  for (PeerFile p : distantFiles)
97  {
98  if (localFiles.contains(p))
99  {
100  int index = localFiles.indexOf(p);
101  localFiles.getReference(index).addPeer(peerIP);
102  }
103  else
104  {
105  p.addPeer(peerIP);
106  localFiles.add(p);
107  }
108  }
109  }
110  }
111  // Else, let him know that something went wrong
112  else
113  {
114  PimpMessage error = PimpMessage::createErrorMessage("Error handling peer refresh");
115  error.sendToSocket(_socket, _owner->getLocalIp());
116  }
